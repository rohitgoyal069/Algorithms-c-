class PalindromeTree {

public:
    vector<ll> s, len, link;
    vector<map<ll, ll>> to;
    string str;
    ll length;
    ll n, last, sz;
    ll distinct;

    void init()
    {
        s = len = link = vector<ll>(length + 10, 0);
        to = vector<map<ll, ll>>(length + 10);
        n = 0, last = 0, sz = distinct = 0;
        s[n++] = -1;
        link[0] = 1;
        len[1] = -1;
        sz = 2;
    }

    PalindromeTree(string s)
    {
        str = s;
        length = s.length();
        init();
        for (auto x : str)
            add_letter(x);
    }

    ll get_link(ll v)
    {
        while (s[n - len[v] - 2] != s[n - 1]) v = link[v];
        return v;
    }

    void add_letter(ll c)
    {
        s[n++] = c;
        last = get_link(last);
        if (!to[last][c])
        {
            len [sz] = len[last] + 2;
            link[sz] = to[get_link(link[last])][c];
            to[last][c] = sz++;
            distinct++;
        }
        last = to[last][c];
    }
};
