ll powa(ll a, ll b, ll c)
{
    a%=c;
    if(a<0)
    a+=c;
    ll res = 1;
    while(b>0)
    {
        if(b&1)
            res*=a, res%=c;
        a*=a, a%=c;
        b>>=1;
    }
    return res;
}

class xorBasis{
    
    public:
    vector<ll> basis;           // basis array
    vector<ll> elem;            // holds original elements used in basis
    ll lim;                     // limit of basis or smallest i such that 2^i > max(elements to be added)
    ll siz;                     // current size of the basis array

    void add(ll ele)
    {
        ll temp = ele;
        for(ll i=0;i<lim;i++)
        {
            ll p2 = (1ll<<i);
            if((p2&ele) == 0)
                continue;  
            if(!basis[i])
            {
                elem.pb(temp);
                basis[i] = ele;
                siz++;
                return;
            }
            
            ele^=basis[i];
        }
    }
    
    void build(vector<ll> arr, ll lim = 63)
    {
        elem.clear();
        siz = 0;
        this->lim = lim;
        basis.resize(lim);
        rep(lim)
            basis[i] = 0;
        ll n = arr.size();
        rep(n)
            if(arr[i]>0)
                add(arr[i]);
    }
    
    bool check(ll num)
    {
        for(ll i=0;i<lim;i++)
        {
            ll p2 = (1ll<<i);
            if((p2&num) == 0)
                continue;
            if(!basis[i])
                return 0;
            p2^=basis[i];    
        }
        return 1;
    }
    
    ll size()
    {
        return siz;
    }
};
